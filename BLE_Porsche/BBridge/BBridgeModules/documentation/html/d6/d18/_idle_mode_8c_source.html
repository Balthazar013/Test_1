101 uint8_t IdleMode_Init(BBridgeAppInterface_t * bbridgeInterface) {
102  uint8_t i;
103  uint8_t ret = BB_SUCCESS;
104  if (bbridgeInterface == BB_SUCCESS) {
105  ret = 2u; /* invalid bbridge interface pointer*/
106  }else if (bbridgeInterface->sharedMemValue == NULL) {
107  ret = 1u; /* invalid shared memory pointer*/
108  }else{
109  /* checking acceptable commands list (it must be sorted)*/
110  for (i = 0u; i < TOTAL_ACCEPTABLE_COMMANDS - 1u; i++) {
111  if (ACCEPTABLE_COMMANDS[i] > ACCEPTABLE_COMMANDS[i + 1u]) {
112  ret = 3u;
113  break;
114  }
115  }
116 
117  if(ret == BB_SUCCESS){
118  controls.bbAppInterface = bbridgeInterface;
119 
120  /* reseting internal working variables*/
121  controls.exitRequested = FALSE;
122  setCMDAsProcessed();
123  controls.state = BBIDLEMD_IDLE;
124  controls.bbAppInterface->sharedMemValueLength = 0u;
125  }
126  }
127  return ret;
128 }

159 uint8_t IdleMode_OnRxCommand(const BBridgeCommand_t * command) {
160  uint8_t ret = BB_SUCCESS;
161  int cmdPosition, searchRet;
162  if (controls.lastCommandProcessed == FALSE) {
163  ret = 1u; /* BUSY*/
164  }else if (command == NULL) {
165  ret = 2u; /* invalid command pointer*/
166  }else{
167 
168  /* check is this mode can accept the command*/
169  searchRet = BBridgeCommons_FindCMDInList(command->id, ACCEPTABLE_COMMANDS,
170  TOTAL_ACCEPTABLE_COMMANDS, &cmdPosition);
171  if (searchRet < 0) {
172  /* command not found in the list*/
173  ret = 3U;
174  }
175 
176  if(ret == BB_SUCCESS){
177  /* ok to accept command*/
178  controls.lastCommand.id = command->id;
179  controls.lastCommand.cmd.length = command->cmd.length;
180  if (controls.lastCommand.cmd.length > 0u) {
181  memcpy(controls.lastCommand.cmd.data, command->cmd.data,
182  (uint16_t)command->cmd.length);
183  }
184  controls.lastCommandProcessed = FALSE; /* setting it to not processed*/
185  }
186  }
187  return ret;

230 void IdleMode_Work(void) {
231  BBCMDType_t cmdType;
232  uint8_t tmp, ret;
233 
234  switch (controls.state) {
235  case (BBIDLEMD_IDLE):
236 
237  if (controls.lastCommandProcessed == FALSE) {
238  /* there is a new command available*/
239 
240  /* deciding how to process received command (checking cmd type)*/
241  cmdType = controls.bbAppInterface->getCMDType(controls.lastCommand.id);
242  switch (cmdType) {
243  case (CMDTYPE_SET):
244  /* Setting values */
245  controls.state = BBIDLEMD_PROCESS_SETCMD;
246  break;
247  case (CMDTYPE_GET):
248  /* Loading values */
249  controls.state = BBIDLEMD_PROCESS_GETCMD;
250  break;
251  default:
252  /* Invalid command -> send NACK */
253  controls.nackReason = BB_NACK_CMD_UNKNOWN;
254  controls.state = BBIDLEMD_SND_NACK;
255  break;
256  }
257  }
258 
259 
260  break;
261  case (BBIDLEMD_PROCESS_GETCMD):
262 
263  tmp = 0U; /* reseting value*/
264 
265  ret = controls.bbAppInterface->proccessGetCmd(&controls.lastCommand,
266  controls.bbAppInterface->sharedMemValue, &tmp);
267  if (ret == BB_SUCCESS) {
268  controls.bbAppInterface->sharedMemValueLength = tmp;
269  controls.state = BBIDLEMD_SND_ACK;
270  } else {
271  /* error processing -> sending NACK*/
272  controls.nackReason = BB_NACK_CMD_UNKNOWN;
273  controls.state = BBIDLEMD_SND_NACK;
274  }
275 
276  break;
277  case (BBIDLEMD_PROCESS_SETCMD):
278 
279  if(controls.lastCommand.id == BBCMD_SET_OP_MODE){
280 
281  /* processing SET OP MODE command */
282  if ( (controls.lastCommand.cmd.data[0] == BB_SCANNER_MODE) ||
283  (controls.lastCommand.cmd.data[0] == BB_CONNECTABLE_MODE)) {
284 
285  controls.newOpMode = (BBridgeOpMode_t) controls.lastCommand.cmd.data[0];
286 
287  /* no need to send ACK at this point (it is done by the BBridge*/
288  /* application module)*/
289 
290  controls.state = BBIDLEMD_WAIT_EXIT_CALL;
291 
292  } else {
293  /* could not set operating mode*/
294  controls.nackReason = BB_NACK_INVALID_REQ;
295  controls.state = BBIDLEMD_SND_NACK;
296  }
297 
298  }else{
299 
300  /* "Set commands" acceptable by the IDLE OP mode can be handled by
301  * the BBridge application mode directly.
302  */
303  ret = controls.bbAppInterface->proccessSetCmd(&controls.lastCommand);
304  if (ret == BB_SUCCESS) {
305  controls.state = BBIDLEMD_SND_ACK;
306  controls.bbAppInterface->sharedMemValueLength = 0U;
307  } else {
308  /* error processing set command, send NACK*/
309  controls.nackReason = BB_NACK_CMD_UNKNOWN;
310  controls.state = BBIDLEMD_SND_NACK;
311  }
312 
313  }
314  break;
315 
316  case (BBIDLEMD_WAIT_TX2PANTHER):
317 
318  if (controls.bbAppInterface->isTxToPantherComplete() == TRUE) {
319  controls.state = BBIDLEMD_IDLE;
320  controls.bbAppInterface->releaseDataHandler(); /* freeing data handler*/
321  setCMDAsProcessed();
322  }
323 
324  break;
325  case (BBIDLEMD_SND_ACK):
326 
327  ret = controls.bbAppInterface->pushACK(controls.lastCommand.id,
328  controls.bbAppInterface->sharedMemValue,
329  controls.bbAppInterface->sharedMemValueLength);
330  if (ret == BB_SUCCESS) {
331  /* data was pushed to be transmitted correctly*/
332  controls.state = BBIDLEMD_WAIT_TX2PANTHER;
333  } else {
334  /* could not push ACK... try again*/
335  }
336 
337  break;
338  case (BBIDLEMD_SND_NACK):
339 
340  ret = controls.bbAppInterface->pushNACK(controls.lastCommand.id,
341  controls.nackReason);
342  if (ret == BB_SUCCESS) {
343  /* data was pushed to be transmitted correctly*/
344  controls.state = BBIDLEMD_WAIT_TX2PANTHER;
345  } else {
346  /* could not push NACK... try again*/
347  }
348  break;
349 
350  case (BBIDLEMD_WAIT_EXIT_CALL):
351 
352  /* Waiting exit "IdleMode_ExitMode" call to be called by the bbridge app */
353  if (controls.exitRequested == TRUE) {
354 
355  /* reseting internal variables*/
356  controls.exitRequested = FALSE;
357  if (controls.lastCommandProcessed == FALSE) {
358  /* there is a command not yet processed (it shall not be possible here)*/
359  controls.lastCommandProcessed = TRUE;
360  }
361 
362  controls.bbAppInterface->releaseDataHandler(); /* freeing data handler*/
363  controls.state = BBIDLEMD_IDLE;
364  }
365 
366  break;
