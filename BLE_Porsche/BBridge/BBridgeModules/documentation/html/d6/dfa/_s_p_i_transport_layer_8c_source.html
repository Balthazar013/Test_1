405 uint8_t SPITransportLayerPushTXData(const uint8_t data[], uint8_t length) {
406  uint8_t ret = BB_SUCCESS;
407  if (attr.txBufferProcessed == FALSE) {
408  ret = 1u;
409  }else if (data == NULL) {
410  ret = 3u; /* invalid data pointer */
411  }else if ((length == 0u) || (length > (SPITRANSPLAYER_BUFFER_SIZE-4u))) {
412  ret = 2u; /* wrong data length */
413  }else if (attr.state != SPIHDLR_IDLE) {
414  /* only allowed to push data when state machine is in IDLE */
415  ret = 4u;
416  }else{
417  attr.txBuffer[0] = PKT_SOF;

420  attr.txBuffer[2] |= PKT_T_DATA;
421

423

493 static frameParserRet_t frameParser(uint8_t rxChar) {
494  frameParserRet_t ret=FP_PARSING;
495  switch (attr.parserState) {
496  case (SPITRNSP_SOF):
497  if (rxChar == PKT_SOF) {
498  /* got a SOF, transition to read PKT length */
499  attr.parserState = SPITRNSP_LEN;
500  } else {
501  ret = FP_IDLE;
502  }
503  break;
504  case (SPITRNSP_LEN):
505  if ((rxChar == 0u) || (rxChar > SPITRANSPLAYER_BUFFER_SIZE)) {
506  /* ERROR reading: Expected valid data size */
507  attr.parserState = SPITRNSP_INVALID;
508  ret = FP_PARSER_FAIL;
509  } else {
510  /* valid length */
511  attr.rxBufferFCS = 0u;
512  attr.rxBufferFCS ^= rxChar;
513  attr.rxBufferLength = (uint8_t) (rxChar - 1u);
514  attr.parserState = SPITRNSP_DATA;
515  attr.rxCounter = 0u;
516  }
517  break;
518  case (SPITRNSP_DATA):
519 
520  attr.rxBufferFCS ^= rxChar;
521  attr.rxBuffer[attr.rxCounter] = rxChar;
522 
523  if (attr.rxCounter == (attr.rxBufferLength - 1u)) {
524  attr.parserState = SPITRNSP_FCS;
525  }
526  attr.rxCounter++;
527 
528  break;
529  case (SPITRNSP_FCS):
530  if (rxChar == attr.rxBufferFCS) {
531  /* data received correctly */
532  attr.parserState = SPITRNSP_DATA_OK;
533  ret = FP_PARSED_OK;
534  } else {
535  /* BAD FCS */
536  attr.parserState = SPITRNSP_INVALID;
537  ret = FP_PARSER_FAIL;
538  }
539  break;
540  case (SPITRNSP_INVALID):
541  ret = FP_PARSER_FAIL;
542  break;
543  case (SPITRNSP_DATA_OK):
544  ret = FP_PARSED_OK;
545  break;
546  default:
547  attr.parserState = SPITRNSP_SOF;
548  break;
549  }
550  return ret;
551 }
