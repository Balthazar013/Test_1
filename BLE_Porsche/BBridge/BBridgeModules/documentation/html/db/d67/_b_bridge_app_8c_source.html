506 static uint8_t setConfigurationValue(const BBridgeCFGId_t cfgID,
507  const uint8_t configValue[], uint8_t length) {
508 
509  uint16_t tmp;
510  uint8_t ret = BB_SUCCESS;
511  uint8_t cfgLength = getCFGDataLength(cfgID);
512  if (cfgLength == 0u) {
513  ret = 1u; /* configuration not valid*/
514  } else if (configValue == NULL) {
515  /* checking if the configuration input is valid*/
516  ret = 4u; /* invalid configuration value*/
517  } else if (cfgID == BBCFG_BROADCAST_MSG) {
518  if (length > BB_MAX_BROADCAST_MSG) {
519  ret = 2u;
520  } else {
521  /* storing value in the BBridgeApp interface*/
522  /* special configuration with 31 bytes*/
523  memset(&controls.broadcastMsg[0], 0, BB_MAX_BROADCAST_MSG);
524  memcpy(&controls.broadcastMsg[0], configValue, (uint16_t) length);
525  controls.configLengths[cfgID] = length;
526 
527  /* setting value in the HW interface*/
528  if ((controls.hwInterface != NULL)
529  && (controls.hwInterface->broadcastMsgSetter != 0)) {
530  if (controls.hwInterface->broadcastMsgSetter(
531  controls.broadcastMsg, length) != BB_SUCCESS) {
532  /* could not set value in the HW interface*/
533  ret = 5u;
534  }
535  }
536  }
537 
538  } else if (cfgID == BBCFG_FILTER_MSG) {
539  if (length > BB_MAX_BROADCAST_MSG) {
540  ret = 2u;
541  } else {
542  memset(&controls.filterMsg[0], 0, BB_MAX_BROADCAST_MSG);
543  if (length == 1u) {
544  /* SPECIAL CASE to set no filter value*/
545  controls.configLengths[cfgID] = 0u;
546  } else {
547  memcpy(controls.filterMsg, configValue, (uint16_t) length);
548  controls.configLengths[cfgID] = length;
549  }
550  }
551  } else {
552 
553  if (length > cfgLength) {
554  ret = 2u; /* configuration value is out of range*/
555  } else {
556  /* normal configuration with max 2 size*/
557  /* coping configuration value*/
558  memcpy(&controls.configValues[cfgID], configValue,
559  (uint16_t) controls.configLengths[cfgID]);
560  if (cfgID == BBCFG_ADV_INTERVAL) {
561  tmp = (uint16_t) (controls.configValues[BBCFG_ADV_INTERVAL][0]);
562  tmp |=
563  (uint16_t) ((uint16_t) controls.configValues[BBCFG_ADV_INTERVAL][1]
564  << 8);
565  if (controls.hwInterface->setActiveTime(tmp) != BB_SUCCESS) {
566  /* could not set value in the HW interface*/
567  ret = 5u;
568  }
569  }
570  }
571  }
572 
573  return ret;
