862 static BBValidResult_t validateConnection(const uint8_t data[], uint8_t length) {
863  uint8_t tmpData[5];
864  BBValidResult_t ret=BBVAL_BUSY;
865 
866  switch (controls.valConnState) {
867  case (BBVLDCONN_IDLE):
868 
869  /* checking if data received from the bluetooth is a SEED request*/
870  if ( (data[0] == BB_SEED_REQ) && (length == 1U) ){
871 
872  /* sending seed response*/
873  controls.valConnSeedGenerated = controls.bbAPPInterface->generateSeed();
874  tmpData[0] = (uint8_t) BB_SEED_RESP; /* seed tx*/
875  tmpData[1] = (uint8_t) (controls.valConnSeedGenerated & 0xFFu);
876  tmpData[2] = (uint8_t) ((controls.valConnSeedGenerated >> 8u) & 0xFFu);
877  tmpData[3] = (uint8_t) ((controls.valConnSeedGenerated >> 16u) & 0xFFu);
878  tmpData[4] = (uint8_t) ((controls.valConnSeedGenerated >> 24u) & 0xFFu);
879 
880  if (controls.bbAPPInterface->sendBTData(controls.currentDevIndex,
881  tmpData, 5u) == BB_SUCCESS) {
882  controls.valConnState = BBVLDCONN_CHECK_KEY;
883  } else {
884  ret = BBVAL_ERROR;
885  }
886  }else{
887  controls.valConnState = BBVLDCONN_AUTH_FAILED;
888  }
889 
890  break;
891  case (BBVLDCONN_CHECK_KEY):
892 
893  if ((data[0] == BB_KEY_RESP) && (length == 5u)) {
894  /* received a key response*/
895  controls.valConnKeyReceived = (uint32_t)data[1];
896  controls.valConnKeyReceived |= ((uint32_t)data[2] << 8u);
897  controls.valConnKeyReceived |= ((uint32_t) data[3] << 16u);
898  controls.valConnKeyReceived |= ((uint32_t) data[4] << 24u);
899 
900  /* checking KEY*/
901  if (controls.valConnKeyReceived
902  == controls.bbAPPInterface->calculateAuthKey(
903  controls.valConnSeedGenerated)
904  ){
905 
906 
907 
908  controls.bbAPPInterface->sharedMemValue[0] = BB_KEY_PASS;
909  if(controls.bbAPPInterface->sendBTData(controls.currentDevIndex,
910  controls.bbAPPInterface->sharedMemValue, 1u) == BB_SUCCESS){
911  controls.valConnState = BBVLDCONN_AUTH_PASS;
912  ret = BBVAL_PASS;
913  }else{
914  controls.valConnState = BBVLDCONN_AUTH_ERROR;
915  ret = BBVAL_ERROR;
916  }
917 
918 
919  } else {
920  controls.valConnState = BBVLDCONN_AUTH_FAILED;
921  ret = BBVAL_FAILED;
922  }
923  } else {
924  controls.valConnState = BBVLDCONN_AUTH_FAILED;
925  ret = BBVAL_FAILED;
926  }
927  break;
928  case (BBVLDCONN_AUTH_PASS):
929  ret = BBVAL_PASS;
930  break;
931  case (BBVLDCONN_AUTH_FAILED):
932  ret = BBVAL_FAILED;
933  break;
934  case(BBVLDCONN_AUTH_ERROR):
935  ret = BBVAL_ERROR;
936  break;
937  default:
938  controls.valConnState = BBVLDCONN_IDLE;
939  break;
940  }
941  return ret;
